```bash
# Run all tests

# Run baseline tests only
npm run test:baseline -- --run

# Run integration tests

npm run test:coverage -- --run

# Run E2E tests
npm run test:e2e

# Update snapshots
npm test -- --run --update
npm run test:e2e -- --update-snapshots
```
